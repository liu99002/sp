//DOWHILE=do STMT while (E)
void DOWHILE(){
  int whileBegin = nextLabel(); //宣告開始標籤
  irEmitLabel(whileBegin); // label (L%d)//開始標籤位置
  skip("do");
  STMT(); //處理迴圈內程式
  skip("while");
  skip("(");
  int e = E(); //處理條件式,並將結果存儲在e
  irEmitIfGoto(e, whileBegin); //如果e符合條件,就回到開始標籤的位置,否則繼續程式
  skip(")");
  skip(";");
}
